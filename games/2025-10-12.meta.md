    ctx.font = "14px sans-serif";
    ctx.fillStyle = "rgba(255,255,255,0.8)";
    const instr = "Press Enter or click Restart to play again.";
    const instrMetrics = safeMeasureText(instr, ctx.font);
    ctx.fillText(instr, (WIDTH - instrMetrics.width) / 2, btnY + btnH + 28);
    ctx.restore();
    // Store restart button rect for input handling
    endButtonRect = { x: btnX, y: btnY, w: btnW, h: btnH };
  }

  let endButtonRect = null;

  function draw(timestamp) {
    clear();
    drawBackgroundWackyMachine(timestamp || 0);
    drawUI();
    if (gameState === "menu") {
      drawMainMenu();
    } else if (gameState === "playing") {
      drawQuestionArea();
      drawChoiceButtons();
    } else if (gameState === "win" || gameState === "lose") {
      drawQuestionArea();
      drawChoiceButtons();
      drawEndScreen();
    }
  }

  function drawMainMenu() {
    ctx.save();
    ctx.fillStyle = "rgba(255,255,255,0.95)";
    const title = "Machine Math";
    ctx.font = "36px sans-serif";
    const metrics = safeMeasureText(title, ctx.font);
    const x = (WIDTH - metrics.width) / 2;
    const y = HEIGHT / 2 - 40;
    roundRect(ctx, x - 20, y - 44, metrics.width + 40, 68, 12, true, false);
    ctx.fillStyle = TEXT_COLOR;
    ctx.fillText(title, x, y);
    ctx.font = "18px sans-serif";
    const text = "Help the wacky machine by answering 10 math questions. 3 mistakes and the machine stops.";
    const lines = wrapTextToLines(text, 520, ctx.font);
    const lineHeight = 22;
    for (let i = 0; i < lines.length; i++) {
      ctx.fillText(lines[i], (WIDTH - safeMeasureText(lines[i], ctx.font).width) / 2, y + 36 + i * lineHeight);
    }
    // Start button
    ctx.font = "20px sans-serif";
    const btnText = "Start";
    const btnMetrics = safeMeasureText(btnText, ctx.font);
    const btnW = btnMetrics.width + 40;
    const btnH = 44;
    const btnX = (WIDTH - btnW) / 2;
    const btnY = y + 120;
    ctx.fillStyle = ACCENT;
    roundRect(ctx, btnX, btnY, btnW, btnH, 10, true, false);
    ctx.fillStyle = "white";
    ctx.fillText(btnText, btnX + (btnW - btnMetrics.width) / 2, btnY + 28);
    ctx.restore();
    // Store start button rect
    endButtonRect = { x: btnX, y: btnY, w: btnW, h: btnH };
  }

  // Input handling
  canvas.addEventListener("click", (e) => {
    const rect = canvas.getBoundingClientRect();
    const mx = e.clientX - rect.left;
    const my = e.clientY - rect.top;
    lastInteractionTime = performance.now();
    // Initialize audio on first user gesture (some browsers require)
    if (!audioCtx) {
      initAudio();
    }
    // If click is within audio icon
    // audio icon located at computed position in drawUI; we can compute similarly
    const livesFont = "18px sans-serif";
    ctx.font = livesFont;
    const livesText = `Faults: ${wrongCount}/${MAX_WRONG}`;
    const livesMetrics = safeMeasureText(livesText, ctx.font);
    const livesW = Math.ceil(livesMetrics.width) + 20;
    const livesX = WIDTH - livesW - UI_PADDING;
    const iconX = livesX - 22 - 8;
    const iconY = 12;
    if (mx >= iconX - 6 && mx <= iconX + 22 && my >= iconY - 6 && my <= iconY + 28) {
      if (audioAvailable) {
        toggleMute();
      } else {
        initAudio();
      }
      return;
    }

    if (gameState === "menu") {
      if (isPointInRect(mx, my, endButtonRect)) {
        startGame();
      }
    } else if (gameState === "playing") {
      // Check choice buttons
      for (const b of buttons) {
        if (isPointInRect(mx, my, b)) {
          handleAnswer(b.index);
          return;
        }
      }
    } else if (gameState === "win" || gameState === "lose") {
      if (isPointInRect(mx, my, endButtonRect)) {
        restartGame();
      }
    }
  });

  function isPointInRect(px, py, rect) {
    if (!rect) return false;
    return px >= rect.x && px <= rect.x + rect.w && py >= rect.y && py <= rect.y + rect.h;
  }

  // Keyboard controls: 1-4 to select answers, Enter to confirm, arrow keys to move, M to mute/unmute, Space/Enter to restart from end.
  canvas.addEventListener("keydown", (e) => {
    lastInteractionTime = performance.now();
    if (!audioCtx) {
      initAudio();
    }
    const key = e.key;
    if (gameState === "menu") {
      if (key === "Enter" || key === " ") {
        startGame();
        e.preventDefault();
      }
    } else if (gameState === "playing") {
      if (["1", "2", "3", "4"].includes(key)) {
        const idx = parseInt(key, 10) - 1;
        handleAnswer(idx);
        e.preventDefault();
      } else if (key === "ArrowLeft" || key === "ArrowUp") {
        keyboardSelection = (keyboardSelection - 1 + buttons.length) % buttons.length;
        e.preventDefault();
      } else if (key === "ArrowRight" || key === "ArrowDown") {
        keyboardSelection = (keyboardSelection + 1) % buttons.length;
        e.preventDefault();
      } else if (key === "Enter") {
        handleAnswer(keyboardSelection);
        e.preventDefault();
      }
    } else if (gameState === "win" || gameState === "lose") {
      if (key === "Enter" || key === " ") {
        restartGame();
        e.preventDefault();
      }
    }
    if (key.toLowerCase() === "m") {
      if (!audioAvailable) initAudio();
      toggleMute();
      e.preventDefault();
    }
  });

  // Also allow global keypresses (when canvas not focused)
  document.addEventListener("keydown", (e) => {
    // forward to canvas handler if not focused
    if (document.activeElement !== canvas) {
      canvas.focus();
    }
  });

  // Visual focus cursor: highlight keyboardSelection
  function updateSelectionHighlight() {
    // simply set selectedIndex to keyboardSelection for drawing
    if (gameState === "playing") {
      selectedIndex = keyboardSelection;
    } else {
      selectedIndex = -1;
    }
  }

  // Animation loop
  function loop(ts) {
    updateSelectionHighlight();
    draw(ts);
    animationId = requestAnimationFrame(loop);
  }

  // Start animation
  animationId = requestAnimationFrame(loop);

  // Ensure canvas gets focus for keyboard accessibility
  canvas.addEventListener("mousedown", () => {
    canvas.focus();
  });

  // Fallback touch support: set focus on touch
  canvas.addEventListener("touchstart", () => {
    canvas.focus();
    if (!audioCtx) initAudio();
  });

  // Start with menu state, offer initAudio on first user gesture
  gameState = "menu";
  // Pre-generate a question for preview in menu so that drawQuestionArea won't error
  generateQuestion();

  // Expose some functions for debugging (in case)
  window.__machineMath = {
    restart: restartGame,
    startAudio: initAudio,
  };

  // Ensure resources are closed on unload
  window.addEventListener("beforeunload", () => {
    try {
      if (bgOsc) bgOsc.stop();
      if (audioCtx && audioCtx.close) audioCtx.close();
    } catch (e) {}
  });

  // Ensure graceful error handling for audio context creation on first user gesture
  // Already handled in initAudio with try/catch

  // Ensure that during drawing, text does not overlap UI elements: we used fixed positions and measured text widths.
  // Place initial focus to canvas for keyboard players
  setTimeout(() => {
    try {
      canvas.focus();
    } catch (e) {}
  }, 100);

})();
