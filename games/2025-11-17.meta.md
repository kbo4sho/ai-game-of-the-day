  // Utility: draw rounded rectangle
  function roundRect(ctx, x, y, w, h, r) {
    const radius = Math.min(r, Math.min(w / 2, h / 2));
    ctx.beginPath();
    ctx.moveTo(x + radius, y);
    ctx.arcTo(x + w, y, x + w, y + h, radius);
    ctx.arcTo(x + w, y + h, x, y + h, radius);
    ctx.arcTo(x, y + h, x, y, radius);
    ctx.arcTo(x, y, x + w, y, radius);
    ctx.closePath();
  }

  // Main loop
  let lastTime = performance.now();
  function loop(now) {
    const dt = now - lastTime;
    lastTime = now;

    // Update
    update(dt);

    // Clear canvas
    ctx.clearRect(0, 0, CANVAS_WIDTH, CANVAS_HEIGHT);

    // Draw
    drawBackground(dt);
    drawOptions();
    drawDrone();
    drawQuestionPanel();
    drawUI();

    // Draw helpful aiming line from drone to nearest option
    ctx.strokeStyle = 'rgba(0,90,120,0.08)';
    ctx.lineWidth = 2;
    const centerX = drone.x;
    let nearest = null;
    let nd = Infinity;
    for (const opt of options) {
      const cx = opt.x + opt.w / 2;
      const d = Math.abs(cx - centerX);
      if (d < nd) {
        nd = d;
        nearest = opt;
      }
    }
    if (nearest) {
      ctx.beginPath();
      ctx.moveTo(centerX, drone.y + drone.height / 2);
      ctx.lineTo(nearest.x + nearest.w / 2, nearest.y);
      ctx.stroke();
      // highlight nearest option background for focus
      ctx.strokeStyle = 'rgba(43,123,191,0.12)';
      ctx.lineWidth = 6;
      ctx.strokeRect(nearest.x + 4, nearest.y + 4, nearest.w - 8, nearest.h - 8);
    }

    // Draw end screen overlay if ended
    if (gameState === 'won' || gameState === 'lost') {
      drawEndScreen();
    }

    requestAnimationFrame(loop);
  }

  function update(dt) {
    if (gameState !== 'playing') return;

    // Move drone based on keys
    if (keysDown.ArrowLeft) {
      drone.vx = -DRONE_SPEED;
    } else if (keysDown.ArrowRight) {
      drone.vx = DRONE_SPEED;
    } else {
      drone.vx = 0;
    }
    drone.x += drone.vx;

    // Boundaries
    const halfW = drone.width / 2;
    if (drone.x < halfW + 8) drone.x = halfW + 8;
    if (drone.x > CANVAS_WIDTH - halfW - 8) drone.x = CANVAS_WIDTH - halfW - 8;

    // gentle bobbing for drone's y when idle
    const bob = Math.sin(performance.now() / 420) * 2;
    if (drone.y !== 140) {
      // if temporarily raised by correct, do not override immediately
    } else {
      drone.y = 140 + bob;
    }
  }

  // Initialize fonts and measurement (ensure minimum font sizes)
  function initFonts() {
    ctx.font = fonts.title;
    ctx.textBaseline = 'top';
  }

  // Start game logic
  initFonts();
  resetGame();
  requestAnimationFrame(loop);

  // Ensure canvas text rendering spacing minimum 10px is respected in draw functions:
  // - We used PADDING = 12 for boxes and ensured UI placement leaves space.

  // Expose a small debug console on the stage element in case of audio error
  if (!audioAvailable) {
    const warnText = document.createElement('div');
    warnText.innerText = 'Audio not available';
    warnText.style.position = 'absolute';
    warnText.style.left = '8px';
    warnText.style.top = '8px';
    warnText.style.background = 'rgba(255,255,255,0.6)';
    warnText.style.padding = '4px 8px';
    warnText.style.borderRadius = '6px';
    warnText.style.fontSize = '12px';
    stage.appendChild(warnText);
  }

  // Error handling examples for audio on resume attempts
  stage.addEventListener('click', () => {
    if (!audioAvailable) return;
    // Try to resume audio context if suspended (some browsers require user gesture)
    if (audioCtx && audioCtx.state === 'suspended') {
      audioCtx.resume().catch((e) => {
        console.warn('AudioContext resume failed:', e);
      });
    }
  });

  // Accessibility: keyboard focus and instructions drawn on canvas; also ensure canvas focus
  canvas.addEventListener('focus', () => {
    // nothing else needed; instructions are shown visually
  });

  // Prevent accidental selection
  canvas.addEventListener('mousedown', (e) => e.preventDefault());

  // Clean up on unload
  window.addEventListener('unload', () => {
    try {
      if (backgroundOsc) backgroundOsc.stop();
      if (audioCtx && typeof audioCtx.close === 'function') audioCtx.close();
    } catch (e) {}
  });
})();
