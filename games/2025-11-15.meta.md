    ctx.fillStyle = '#1565c0';
    roundRect(ctx, SCORE_X + 2, SCORE_Y + 2, scoreW - 4, scoreH - 4, 6);
    ctx.fill();
    ctx.fillStyle = '#fff';
    ctx.fillText(scoreText, SCORE_X + PADDING, SCORE_Y + 8);

    // Lives top-right
    ctx.font = IMPORTANT_FONT;
    const livesText = `Fails left: ${Math.max(0, GameState.maxWrong - GameState.wrong)}`;
    const livesMetrics = ctx.measureText(livesText);
    const livesW = livesMetrics.width + 2 * PADDING + 44; // space for audio icon
    const livesH = 34;
    const livesX = LIVES_X - livesW;
    ctx.fillStyle = 'rgba(255,255,255,0.9)';
    roundRect(ctx, livesX, LIVES_Y, livesW, livesH, 8);
    ctx.fillStyle = '#f48fb1';
    roundRect(ctx, livesX + 2, LIVES_Y + 2, livesW - 4, livesH - 4, 6);
    ctx.fill();
    ctx.fillStyle = '#fff';
    ctx.fillText(livesText, livesX + PADDING, LIVES_Y + 8);

    // Audio icon and status
    const audRect = getAudioIconRect();
    ctx.fillStyle = audio.enabled ? '#fff' : '#fff';
    // small background
    ctx.fillStyle = 'rgba(255,255,255,0.06)';
    ctx.fillRect(audRect.x, audRect.y, audRect.w, audRect.h);
    ctx.strokeStyle = '#fff';
    ctx.lineWidth = 1;
    // draw speaker icon
    ctx.beginPath();
    ctx.moveTo(audRect.x + 3, audRect.y + 5);
    ctx.lineTo(audRect.x + 9, audRect.y + 5);
    ctx.lineTo(audRect.x + 13, audRect.y + 2);
    ctx.lineTo(audRect.x + 13, audRect.y + 18);
    ctx.lineTo(audRect.x + 9, audRect.y + 15);
    ctx.lineTo(audRect.x + 3, audRect.y + 15);
    ctx.closePath();
    ctx.fillStyle = audio.enabled ? '#fff' : '#aaa';
    ctx.fill();
    if (!audio.enabled) {
      ctx.strokeStyle = '#ff8a80';
      ctx.beginPath();
      ctx.moveTo(audRect.x + 2, audRect.y + 2);
      ctx.lineTo(audRect.x + audRect.w - 2, audRect.y + audRect.h - 2);
      ctx.stroke();
    }

    // Draw the question area centrally
    ctx.font = TITLE_FONT;
    ctx.textBaseline = 'top';
    const qText = `What is ${GameState.question.text}?`;
    const qMetrics = ctx.measureText(qText);
    const qW = Math.min(WIDTH - 2 * PADDING, qMetrics.width + 2 * PADDING);
    const qX = (WIDTH - qW) / 2;
    const qY = 170;
    ctx.fillStyle = 'rgba(255,255,255,0.95)';
    roundRect(ctx, qX, qY, qW, 52, 10);
    ctx.fillStyle = '#0277bd';
    roundRect(ctx, qX + 2, qY + 2, qW - 4, 48, 8);
    ctx.fill();
    ctx.fillStyle = '#fff';
    ctx.font = IMPORTANT_FONT;
    // ensure text doesn't overflow: measure and if too wide reduce font
    let fontToUse = IMPORTANT_FONT;
    ctx.font = fontToUse;
    let textWidth = ctx.measureText(qText).width;
    if (textWidth > qW - 2 * PADDING) {
      fontToUse = '18px "Segoe UI", Roboto, Arial';
      ctx.font = fontToUse;
      textWidth = ctx.measureText(qText).width;
    }
    ctx.fillText(qText, qX + PADDING, qY + 12);

    // Draw options below, spaced and measured to avoid overlap
    const optionY = qY + 72;
    const areaPadding = 12;
    optionAreas = [];
    // compute equal widths but also use measureText to ensure enough
    ctx.font = BODY_FONT;
    let widths = GameState.question.options.map(opt => ctx.measureText(String(opt)).width + 2 * areaPadding + 40);
    // minimum width and spacing
    const totalMinWidth = widths.reduce((a, b) => a + b, 0);
    const gap = Math.max(12, Math.floor((WIDTH - 2 * PADDING - totalMinWidth) / (OPTION_COUNT + 1)));
    let curX = PADDING + gap;
    for (let i = 0; i < GameState.question.options.length; i++) {
      const opt = GameState.question.options[i];
      const w = widths[i];
      const h = 52;
      // draw background
      const selected = (GameState.selectedIndex === i);
      ctx.fillStyle = selected ? 'rgba(255, 243, 224, 0.95)' : 'rgba(255,255,255,0.95)';
      roundRect(ctx, curX, optionY, w, h, 8);
      ctx.fillStyle = selected ? '#ff8f00' : '#607d8b';
      roundRect(ctx, curX + 2, optionY + 2, w - 4, h - 4, 6);
      ctx.fill();
      // number label
      ctx.fillStyle = '#fff';
      ctx.font = 'bold 16px Arial';
      ctx.fillText(`${i + 1}.`, curX + 12, optionY + 14);
      ctx.font = BODY_FONT;
      ctx.fillText(String(opt), curX + 44, optionY + 14);

      optionAreas.push({ x: curX, y: optionY, w, h, index: i });
      curX += w + gap;
    }

    // Instructions bottom-center
    ctx.font = BODY_FONT;
    ctx.textBaseline = 'top';
    const instructions = 'Use arrow keys or 1-3 to choose. Press Enter to confirm. Press M to toggle sound. Press R to restart.';
    const maxWidth = WIDTH - 2 * PADDING;
    // split instructions into two lines if too wide
    ctx.font = BODY_FONT;
    const instrWidth = ctx.measureText(instructions).width;
    let instrLines = [];
    if (instrWidth > maxWidth) {
      // naive split near middle space
      const words = instructions.split(' ');
      let line = '';
      for (let w of words) {
        const trial = line ? (line + ' ' + w) : w;
        if (ctx.measureText(trial).width > maxWidth && line) {
          instrLines.push(line);
          line = w;
        } else {
          line = trial;
        }
      }
      if (line) instrLines.push(line);
    } else {
      instrLines = [instructions];
    }
    // draw background rectangle
    const instrH = instrLines.length * 20 + 12;
    const instrW = maxWidth;
    const instrX = (WIDTH - instrW) / 2;
    const instrY = INSTRUCTIONS_Y;
    ctx.fillStyle = 'rgba(255,255,255,0.9)';
    roundRect(ctx, instrX, instrY, instrW, instrH, 10);
    ctx.fillStyle = '#455a64';
    roundRect(ctx, instrX + 2, instrY + 2, instrW - 4, instrH - 4, 8);
    ctx.fill();
    ctx.fillStyle = '#fff';
    for (let i = 0; i < instrLines.length; i++) {
      ctx.fillText(instrLines[i], instrX + PADDING, instrY + 8 + i * 20);
    }

    // If the game is won or over, overlay victory/gameover screens
    if (GameState.state === 'victory') {
      drawEndScreen(true);
    } else if (GameState.state === 'gameover') {
      drawEndScreen(false);
    }

    // draw small help for audio availability if audio failed
    if (!audio.isAvailable()) {
      ctx.font = '14px Arial';
      const warn = 'Audio unavailable. Press M to toggle (visual only).';
      const m = ctx.measureText(warn).width;
      ctx.fillStyle = 'rgba(255,255,255,0.9)';
      roundRect(ctx, WIDTH - m - 2 * PADDING - 8, HEIGHT - 34, m + 2 * PADDING + 8, 30, 6);
      ctx.fillStyle = '#6d4c41';
      roundRect(ctx, WIDTH - m - 2 * PADDING - 6, HEIGHT - 32, m + 2 * PADDING + 4, 26, 5);
      ctx.fill();
      ctx.fillStyle = '#fff';
      ctx.fillText(warn, WIDTH - m - PADDING - 4, HEIGHT - 28);
    }
  }

  function drawEndScreen(victory) {
    // overlay a translucent panel
    ctx.fillStyle = 'rgba(5, 20, 40, 0.65)';
    ctx.fillRect(0, 0, WIDTH, HEIGHT);

    // main box
    const boxW = WIDTH - 120;
    const boxH = 220;
    const boxX = 60;
    const boxY = (HEIGHT - boxH) / 2;
    ctx.fillStyle = 'rgba(255,255,255,0.95)';
    roundRect(ctx, boxX, boxY, boxW, boxH, 12);
    ctx.fillStyle = victory ? '#2e7d32' : '#b71c1c';
    roundRect(ctx, boxX + 4, boxY + 4, boxW - 8, boxH - 8, 10);
    ctx.fill();

    ctx.fillStyle = '#fff';
    ctx.font = '28px "Segoe UI", Roboto, Arial';
    const title = victory ? 'You did it! Drone delivered the stars!' : 'Oh no! The drone ran out of battery!';
    const titleWidth = ctx.measureText(title).width;
    ctx.fillText(title, boxX + (boxW - titleWidth) / 2, boxY + 26);

    ctx.font = '18px "Segoe UI", Roboto, Arial';
    const sub = victory ? `You collected ${GameState.score} stars!` : `You collected ${GameState.score} stars.`;
    const subWidth = ctx.measureText(sub).width;
    ctx.fillText(sub, boxX + (boxW - subWidth) / 2, boxY + 70);

    ctx.font = '16px Arial';
    const instr = 'Click anywhere or press R to restart.';
    const instrW = ctx.measureText(instr).width;
    ctx.fillText(instr, boxX + (boxW - instrW) / 2, boxY + boxH - 46);

    // friendly drone drawing on the end screen
    drawDrone(boxX + 100, boxY + 130, GameState.wobble, true);
  }

  // Error handling for audio context on user gesture failure
  // Try to initialize audio on first pointerdown as well (some browsers require gesture)
  function tryInitAudioOnGesture() {
    audio.init().then(() => {
      // resume if suspended
      try { if (audio.context && audio.context.state === 'suspended') audio.context.resume(); }
      catch (e) { /* ignore */ }
    }).catch(() => {});
    window.removeEventListener('pointerdown', tryInitAudioOnGesture);
    window.removeEventListener('keydown', tryInitAudioOnGesture);
  }
  window.addEventListener('pointerdown', tryInitAudioOnGesture);
  window.addEventListener('keydown', tryInitAudioOnGesture);
})();
