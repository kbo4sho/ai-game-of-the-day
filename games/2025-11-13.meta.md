    const bh = 120;
    ctx.beginPath();
    ctx.moveTo(x - 8, y + h / 2);
    ctx.lineTo(x + 8, y + h / 2);
    ctx.lineTo(x + bw / 2, y + h / 2 + bh);
    ctx.lineTo(x - bw / 2, y + h / 2 + bh);
    ctx.closePath();
    ctx.fill();
    ctx.restore();
  }

  // --- Main render loop ---
  function update(dt) {
    if (!state.running) {
      // subtle idle animation
      state.drone.x += Math.sin(Date.now() / 800) * 0.2;
      return;
    }
    // Movement based on keys and mouse
    let targetVX = 0;
    if (state.keys.left) targetVX -= 1;
    if (state.keys.right) targetVX += 1;
    // Also keyboard up/down adjust vertical movement slightly
    let targetVY = 0;
    if (state.keys.up) targetVY -= 1;
    if (state.keys.down) targetVY += 1;

    // Smooth velocity
    const accel = 10;
    state.drone.vx += (targetVX * state.drone.speed - state.drone.vx) * Math.min(1, accel * dt);
    state.drone.vy += (targetVY * state.drone.speed - state.drone.vy) * Math.min(1, accel * dt);

    // Mouse guiding if mouse near top half: subtle attraction to mouse x
    if (canvas === document.activeElement) {
      // only when canvas focused
      const dx = state.mouse.x - state.drone.x;
      state.drone.vx += dx * 0.002;
    }

    // Update positions
    state.drone.x += state.drone.vx * dt;
    state.drone.y += state.drone.vy * dt;

    // Boundaries
    state.drone.x = clamp(state.drone.x, state.drone.width / 2 + 10, canvas.width - state.drone.width / 2 - 10);
    state.drone.y = clamp(state.drone.y, 60, CONFIG.crateY - 70);

    // Update crate animations if picked
    for (const crate of state.crates) {
      if (crate.picked) {
        crate.y -= 60 * dt;
        crate.w *= 0.995;
        crate.h *= 0.995;
      }
    }

    // Flash timer reduce
    if (state.flashTimer > 0) state.flashTimer = Math.max(0, state.flashTimer - dt);
  }

  function draw() {
    // Clear background - calming gradient sky
    const grad = ctx.createLinearGradient(0, 0, 0, canvas.height);
    grad.addColorStop(0, '#E8F6FF');
    grad.addColorStop(0.6, '#F7FFF7');
    ctx.fillStyle = grad;
    ctx.fillRect(0, 0, canvas.width, canvas.height);

    // Decorative wacky elements - floating balloons/drones
    for (let i = 0; i < 6; i++) {
      const bx = (i * 137 + Date.now() / 40) % canvas.width;
      const by = 40 + (i % 3) * 16 + Math.sin((Date.now() / 1000) + i) * 8;
      ctx.globalAlpha = 0.25;
      ctx.fillStyle = ['#FFDDE2', '#E0FFE6', '#E9F5FF'][i % 3];
      ctx.beginPath();
      ctx.ellipse(bx, by, 22, 12, 0, 0, Math.PI * 2);
      ctx.fill();
      ctx.globalAlpha = 1;
    }

    // UI Layout with strict non-overlapping positions
    // Score top-left
    ctx.font = `${CONFIG.importantFont}px sans-serif`;
    const scoreText = `Score: ${state.score}/${CONFIG.goalCorrect}`;
    const scoreBox = drawTextBox(scoreText, CONFIG.padding, CONFIG.padding, {
      align: 'left',
      fontSize: CONFIG.importantFont,
      padding: 12,
      bg: 'rgba(255,255,255,0.9)',
      fg: '#0A2749',
    });

    // Audio icon top-center
    const audioText = state.audioOn ? 'Audio: On (M to toggle)' : 'Audio: Off (M to toggle)';
    ctx.font = `${16}px sans-serif`;
    const audioMetrics = ctx.measureText(audioText);
    const audioBoxW = audioMetrics.width + 16 + 24; // icon width
    const audioX = canvas.width / 2 - audioBoxW / 2;
    const audioY = CONFIG.padding;
    // background
    ctx.fillStyle = 'rgba(255,255,255,0.9)';
    roundRect(ctx, audioX, audioY, audioBoxW, 36, 8, true, false);
    // icon
    ctx.fillStyle = state.audioOn ? '#2E8B57' : '#B23A48';
    ctx.fillRect(audioX + 6, audioY + 8, 20, 20);
    ctx.fillStyle = '#06283D';
    ctx.font = '14px sans-serif';
    ctx.textBaseline = 'middle';
    ctx.fillText(audioText, audioX + 34, audioY + 18);

    // Lives top-right
    ctx.font = `${CONFIG.importantFont}px sans-serif`;
    const livesLeft = Math.max(0, CONFIG.maxWrong - state.wrong);

    const livesMetrics = ctx.measureText(livesText);
    const livesBoxW = livesMetrics.width + 24;

    roundRect(ctx, livesX, livesY, livesBoxW, 36, 8, true, false);
    ctx.fillStyle = '#06283D';
    ctx.fillText(livesText, livesX + 12, livesY + 22);

    const qText = state.question ? `Question: ${state.question.a} ${state.question.op} ${state.question.b} = ?` : 'Loading...';
    const qBox = drawTextBox(qText, canvas.width / 2, 70, {
      align: 'center',
      fontSize: 20,
      padding: 12,
      bg: 'rgba(255,255,255,0.95)',
      fg: '#06283D',

    drawDrone();

    for (const crate of state.crates) {

    }

    const instr = 'Controls: Arrow keys to move, Space or Enter to scan a crate. Click a crate to move over it. Reach 10 correct answers. Press M to toggle audio.';
    ctx.font = `${CONFIG.minBodyFont}px sans-serif`;
    // Break into lines to avoid overlap. We'll wrap roughly to width 640.
    const wrapWidth = 640;
    const words = instr.split(' ');
    let line = '';
    const lines = [];
    for (const word of words) {
      const test = line ? line + ' ' + word : word;
      const w = ctx.measureText(test).width;
      if (w > wrapWidth) {
        lines.push(line);
        line = word;
      } else {
        line = test;

      ctx.fillRect(0, 0, canvas.width, canvas.height);
      ctx.globalAlpha = 1;

    if (!state.running && (state.win || state.lose)) {

      const boxW = Math.max(titleMetrics.width, msgMetrics.width) + 80;

      const by2 = canvas.height / 2 - boxH / 2;

      roundRect(ctx, bx2, by2, boxW, boxH, 12, true, false);

      ctx.font = '36px sans-serif';
      ctx.textBaseline = 'top';
      ctx.fillText(title, canvas.width / 2 - ctx.measureText(title).width / 2, by2 + 16);
      ctx.font = '20px sans-serif';
      ctx.fillText(msg, canvas.width / 2 - ctx.measureText(msg).width / 2, by2 + 64);
      // Restart button
      const btnW = 180;
      const btnH = 44;
      const bxbtn = canvas.width / 2 - btnW / 2;
      const bybtn = by2 + boxH - 56;
      ctx.fillStyle = '#2E8B57';
      roundRect(ctx, bxbtn, bybtn, btnW, btnH, 8, true, false);
      ctx.fillStyle = '#fff';
      ctx.font = '20px sans-serif';
      ctx.fillText('Restart (R)', canvas.width / 2 - ctx.measureText('Restart (R)').width / 2, bybtn + 12);
    }

    // Ensure alpha reset
    ctx.globalAlpha = 1;
  }

  // --- Main loop ---
  function loop(timestamp) {
    if (!state.lastTime) state.lastTime = timestamp;
    const dt = Math.min(0.05, (timestamp - state.lastTime) / 1000); // clamp dt
    state.lastTime = timestamp;

    update(dt);
    draw();

    requestAnimationFrame(loop);
  }

  // Start game
  try {
    newQuestion();
    if (state.audioOn && audioEnabled && audioCtx) {
      // Start background hum when user interacts; resume on initial click or keypress
      // We'll try to resume immediately; browsers require user gesture, so we attach one-time handler.
      function resumeAudioOnGesture() {
        if (audioCtx && audioCtx.state === 'suspended') {
          audioCtx.resume().then(() => {
            startBackgroundHum();
          }).catch(() => {
            // ignore
          });
        } else {
          startBackgroundHum();
        }
        window.removeEventListener('pointerdown', resumeAudioOnGesture);
        window.removeEventListener('keydown', resumeAudioOnGesture);
      }
      window.addEventListener('pointerdown', resumeAudioOnGesture);
      window.addEventListener('keydown', resumeAudioOnGesture);
    }
    requestAnimationFrame(loop);
  } catch (e) {
    console.error('Game start error:', e);
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    ctx.fillStyle = '#fff';
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    ctx.fillStyle = '#000';
    ctx.font = '18px sans-serif';
    ctx.fillText('An error occurred starting the game. Please try reloading.', 20, 40);
  }

  // Expose restart to console for debugging
  window.__droneMathRestart = restartGame;
})();
